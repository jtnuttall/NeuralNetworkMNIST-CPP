12 int reverseInt(int i)
13 {
14  unsigned char ch1, ch2, ch3, ch4;
15  ch1 = i & 255;
16  ch2 = (i >> 8) & 255;
17  ch3 = (i >> 16) & 255;
18  ch4 = (i >> 24) & 255;
19  return((int)ch1 << 24) + ((int)ch2 << 16) + ((int)ch3 << 8) + ch4;
20 }

38  n_cols = reverseInt(n_cols);
39 
40  images.resize(number_of_images);
41  for (int i = 0; i < number_of_images; ++i)
42  {
43  images[i].resize(n_rows * n_cols);
44  for (int r = 0; r < n_rows; ++r)
45  {
46  for (int c = 0; c < n_cols; ++c)
47  {
48  unsigned char pixel = 0;
49  file.read((char*)&pixel, sizeof(pixel));
50  images[i][n_rows * r + c] = (int)pixel;
51  }
52  }
53  }
